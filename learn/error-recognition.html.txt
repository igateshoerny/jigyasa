        // Function for error recognition practice
        function checkErrorAnswer(questionId, correctAnswer) {
            const options = document.querySelectorAll(`#${questionId} .error-option`);
            const resultDiv = document.getElementById(`${questionId}-result`);
            
            let selectedOption = null;
            options.forEach(option => {
                if (option.classList.contains('selected')) {
                    selectedOption = option.getAttribute('data-value');
                }
            });

            if (!selectedOption) {
                resultDiv.textContent = "Pilih bagian yang salah terlebih dahulu!";
                resultDiv.className = "quiz-result incorrect";
                resultDiv.style.display = 'block';
                return;
            }

            if (selectedOption === correctAnswer) {
                resultDiv.innerHTML = "‚úÖ <strong>Benar!</strong> Kamu berhasil menemukan errornya!";
                resultDiv.className = "quiz-result correct";
            } else {
                resultDiv.innerHTML = `‚ùå <strong>Masih salah!</strong> Error sebenarnya ada di bagian: ${correctAnswer}`;
                resultDiv.className = "quiz-result incorrect";
            }
            
            resultDiv.style.display = 'block';
        }

        // Add click event to all error options
        document.querySelectorAll('.error-option').forEach(option => {
            option.addEventListener('click', function() {
                // Remove selection from other options in the same question
                const parentOptions = this.closest('.error-options');
                parentOptions.querySelectorAll('.error-option').forEach(opt => {
                    opt.classList.remove('selected');
                });
                
                this.classList.add('selected');
            });
        });
