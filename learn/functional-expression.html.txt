            <div class="lesson-section" id="making-requests">
                <h2>🙏 Making Requests</h2>
                
                <h3>Formal Requests</h3>
                <div class="expression-card">
                    <h4>Could you...?</h4>
                    <p><strong>Example:</strong> "Could you help me carry this box?"</p>
                    <p><strong>Appropriate Response:</strong> "Yes, with pleasure." / "Of course."</p>
                </div>
                
                <div class="expression-card">
                    <h4>Would you mind...?</h4>
                    <p><strong>Example:</strong> "Would you mind opening the window?"</p>
                    <p><strong>Appropriate Response:</strong> "Of course not." / "Not at all."</p>
                </div>

                <h3>Informal Requests</h3>
                <div class="expression-card">
                    <h4>Can you...?</h4>
                    <p><strong>Example:</strong> "Can you pass me the salt?"</p>
                    <p><strong>Appropriate Response:</strong> "Sure, here you are."</p>
                </div>

                <div class="tip-box">
                    <strong>💡 Politeness Scale:</strong><br>
                    • <strong>Most polite:</strong> "Would you mind...?"<br>
                    • <strong>Polite:</strong> "Could you...?"<br>
                    • <strong>Casual:</strong> "Can you...?"
                </div>
            </div>

            <!-- Asking Permission -->
            <div class="lesson-section" id="asking-permission">
                <h2>📋 Asking Permission</h2>
                
                <div class="expression-card">
                    <h4>May I...? (Formal)</h4>
                    <p><strong>Example:</strong> "May I sit here?"</p>
                    <p><strong>Appropriate Response:</strong> "Of course, please do." / "Yes, certainly."</p>
                </div>
                
                <div class="expression-card">
                    <h4>Can I...? (Informal)</h4>
                    <p><strong>Example:</strong> "Can I borrow your pen?"</p>
                    <p><strong>Appropriate Response:</strong> "Yes, here you are." / "Sure, no problem."</p>
                </div>

                <div class="expression-card">
                    <h4>Do you mind if I...?</h4>
                    <p><strong>Example:</strong> "Do you mind if I open the window?"</p>
                    <p><strong>Appropriate Response:</strong> "No, not at all." / "Go ahead."</p>
                </div>

            <div class="lesson-section" id="giving-suggestions">
                <h2>💡 Giving Suggestions</h2>
                
                <div class="expression-card">
                    <h4>How about...?</h4>
                    <p><strong>Example:</strong> "How about going to the beach this weekend?"</p>
                    <p><strong>Appropriate Response:</strong> "That's a good idea!" / "Sounds great!"</p>
                </div>
                
                <div class="expression-card">
                    <h4>Why don't we...?</h4>
                    <p><strong>Example:</strong> "Why don't we study together?"</p>
                    <p><strong>Appropriate Response:</strong> "That would be helpful." / "Good idea."</p>
                </div>

                <div class="expression-card">
                    <h4>Let's...</h4>
                    <p><strong>Example:</strong> "Let's go out for dinner."</p>
                    <p><strong>Appropriate Response:</strong> "Sure, let's do that." / "Okay, why not?"</p>
                </div>

                <div class="situation-box">
                    <strong>Common Mistake:</strong><br>
                    ❌ "How about <em>to go</em> to the beach?" (salah)<br>
                    ✅ "How about <em>going</em> to the beach?" (benar)<br><br>
                    
                    <strong>Rule:</strong> After "How about" use <strong>gerund (-ing)</strong>
                </div>

            <!-- Context Understanding -->
            <div class="lesson-section" id="context-understanding">
                <h2>🔍 Context Understanding</h2>
                
                <h3>Key Strategies</h3>
                <div class="detector-box">
                    <strong>1. Identify the Situation</strong><br>
                    • Formal vs informal setting<br>
                    • Relationship between speakers<br>
                    • Location and context<br><br>

                    <strong>2. Recognize the Speech Act</strong><br>
                    • Requesting vs offering<br>
                    • Complaining vs complimenting<br>
                    • Informing vs asking<br><br>

                    <strong>3. Match the Tone</strong><br>
                    • Formal question → formal response<br>
                    • Casual suggestion → casual agreement<br>
                    • Emotional sharing → empathetic response
                </div>

                <div class="tip-box">
                    <strong>🎯 Final Tips for Functional Expression:</strong><br>
                    1. <strong>Perhatikan tingkat formalitas</strong> percakapan<br>
                    2. <strong>Pahami tujuan</strong> dari ungkapan yang diberikan<br>
                    3. <strong>Pilih respons yang sesuai</strong> dengan konteks sosial<br>
                    4. <strong>Hindari terjemahan literal</strong> dari bahasa Indonesia<br>
                    5. <strong>Pelajari pola</strong> respons yang umum digunakan native speakers
                </div>

    <script>
        // Function for response practice
        function checkResponseAnswer(questionId, correctAnswer) {
            const options = document.querySelectorAll(`#${questionId} .response-option`);
            const resultDiv = document.getElementById(`${questionId}-result`);
            
            let selectedOption = null;
            options.forEach(option => {
                if (option.classList.contains('selected')) {
                    selectedOption = option.getAttribute('data-value');
                }
            });

            if (!selectedOption) {
                resultDiv.textContent = "Pilih respons terlebih dahulu!";
                resultDiv.className = "quiz-result incorrect";
                resultDiv.style.display = 'block';
                return;
            }

            if (selectedOption === correctAnswer) {
                resultDiv.innerHTML = "✅ <strong>Benar!</strong> Kamu memilih respons yang appropriate!";
                resultDiv.className = "quiz-result correct";
            } else {
                resultDiv.innerHTML = `❌ <strong>Masih salah!</strong> Respons yang appropriate adalah: ${correctAnswer}`;
                resultDiv.className = "quiz-result incorrect";
            }
            
            resultDiv.style.display = 'block';
        }

        // Add click event to all response options
        document.querySelectorAll('.response-option').forEach(option => {
            option.addEventListener('click', function() {
                // Remove selection from other options in the same question
                const parentOptions = this.closest('.response-options');
                parentOptions.querySelectorAll('.response-option').forEach(opt => {
                    opt.classList.remove('selected');
                });
                
                this.classList.add('selected');
            });
        });

        // Smooth scrolling for table of contents
        document.querySelectorAll('.toc-list a').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                const targetElement = document.querySelector(targetId);
                targetElement.scrollIntoView({ behavior: 'smooth' });
            });
        });
    </script>
