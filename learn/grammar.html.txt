            <!-- 16 Tenses -->
            <div class="lesson-section" id="tenses">
                <h2>⏰ 16 Tenses Mastery</h2>
                
                <h3>Simple Present</h3>
                <p><strong>Rumus:</strong> S + V1 (s/es)</p>
                <p><strong>Penggunaan:</strong> Kebiasaan, fakta, jadwal</p>
                <div class="example-box correct">
                    <strong>Contoh:</strong> She <em>goes</em> to school every day.
                </div>

                <h3>Present Continuous</h3>
                <p><strong>Rumus:</strong> S + am/is/are + V-ing</p>
                <p><strong>Penggunaan:</strong> Aksi sedang berlangsung sekarang</p>
                <div class="example-box correct">
                    <strong>Contoh:</strong> She <em>is going</em> to school by bus today.
                </div>

                <h3>Present Perfect</h3>
                <p><strong>Rumus:</strong> S + have/has + V3</p>
                <p><strong>Penggunaan:</strong> Pengalaman, aksi yang berpengaruh sampai sekarang</p>
                <div class="example-box correct">
                    <strong>Contoh:</strong> I <em>have finished</em> my homework.
                </div>

                <div class="tip-box">
                    <strong>💡 Tips Olimpiade:</strong> Perhatikan kata kunci! <br>
                    • <em>every day, always</em> → Simple Present<br>
                    • <em>now, at the moment</em> → Present Continuous<br>
                    • <em>already, since, for</em> → Present Perfect
                </div>

                <table>
                    <thead>
                        <tr>
                            <th>Tense</th>
                            <th>Rumus</th>
                            <th>Kata Kunci</th>
                        </tr>
                    </thead>
                    <tbody>
                        <tr>
                            <td>Simple Past</td>
                            <td>S + V2</td>
                            <td>yesterday, last week, ago</td>
                        </tr>
                        <tr>
                            <td>Past Continuous</td>
                            <td>S + was/were + V-ing</td>
                            <td>while, when, at 8 PM yesterday</td>
                        </tr>
                        <tr>
                            <td>Past Perfect</td>
                            <td>S + had + V3</td>
                            <td>before, after, by the time</td>
                        </tr>
                    </tbody>
                </table>
            </div>

            <!-- Conditional Sentences -->
            <div class="lesson-section" id="conditional">
                <h2>🔀 Conditional Sentences</h2>
                
                <h3>Type 0 - Fakta Umum</h3>
                <p><strong>Rumus:</strong> If + present simple, present simple</p>
                <div class="example-box correct">
                    If you <em>heat</em> ice, it <em>melts</em>.
                </div>

                <h3>Type 1 - Real Possibility</h3>
                <p><strong>Rumus:</strong> If + present simple, will + V1</p>
                <div class="example-box correct">
                    If it <em>rains</em> tomorrow, we <em>will cancel</em> the picnic.
                </div>

                <h3>Type 2 - Unreal Present</h3>
                <p><strong>Rumus:</strong> If + past simple, would + V1</p>
                <div class="example-box correct">
                    If I <em>were</em> rich, I <em>would travel</em> around the world.
                </div>

                <h3>Type 3 - Unreal Past</h3>
                <p><strong>Rumus:</strong> If + past perfect, would have + V3</p>
                <div class="example-box correct">
                    If I <em>had known</em> about the meeting, I <em>would have gone</em> there.
                </div>

                <div class="tip-box">
                    <strong>🚨 Common Mistake:</strong> Hati-hati dengan subjunctive!<br>
                    <div class="example-box incorrect">
                        ❌ If I <em>was</em> you... (salah)
                    </div>
                    <div class="example-box correct">
                        ✅ If I <em>were</em> you... (benar)
                    </div>
                </div>
            </div>

            <!-- Passive Voice -->
            <div class="lesson-section" id="passive">
                <h2>🔄 Passive Voice</h2>
                
                <p><strong>Rumus Umum:</strong> S + be + V3 + (by agent)</p>

                <table>
                    <thead>
                        <tr>
                            <th>Tense</th>
                            <th>Active</th>

                        <tr>
                            <td>Past Simple</td>
                            <td>She wrote a letter</td>
                            <td>A letter <strong>was written</strong> by her</td>
                        </tr>
                        <tr>
                            <td>Present Perfect</td>
                            <td>She has written a letter</td>
                            <td>A letter <strong>has been written</strong> by her</td>
                        </tr>
                    </tbody>
                </table>

                <div class="example-box correct">
                    <strong>Contoh Soal Olimpiade:</strong><br>
                    English ______ in many countries.<br>
                    ✅ <em>is spoken</em> (benar)<br>
                    ❌ speak, is speaking, spoken (salah)

            <!-- Subject-Verb Agreement -->
            <div class="lesson-section" id="subject-verb">
                <h2>📝 Subject-Verb Agreement</h2>
                
                <h3>Aturan Dasar</h3>
                <ul>
                    <li>Singular subject → singular verb (+s/es)</li>
                    <li>Plural subject → plural verb (tanpa s/es)</li>
                </ul>

                <div class="example-box incorrect">
                    ❌ She <em>don't</em> like coffee. (salah)
                </div>
                <div class="example-box correct">
                    ✅ She <em>doesn't</em> like coffee. (benar)
                </div>

                <h3>Special Cases</h3>
                <p><strong>Either...or / Neither...nor:</strong> Verb mengikuti subjek terdekat</p>
                <div class="example-box correct">
                    Neither my friends nor I <em>am</em> going to the party.
                </div>

                <p><strong>Along with, together with:</strong> Verb mengikuti subjek utama</p>
                <div class="example-box correct">
                    The man, along with his children, <em>is</em> going to the park.
                </div>
            </div>

            <!-- Causative Verbs -->
            <div class="lesson-section" id="causative">
                <h2>⚡ Causative Verbs</h2>
                
                <h3>Make + object + base verb</h3>
                <div class="example-box correct">
                    The teacher made the students <em>clean</em> the classroom.
                </div>

                <h3>Have + object + base verb</h3>
                <div class="example-box correct">
                    I had my brother <em>fix</em> my computer.
                </div>

                <h3>Get + object + to-infinitive</h3>
                <div class="example-box correct">
                    She got her friend <em>to help</em> with the project.
                </div>

                <div class="tip-box">
                    <strong>💡 Perbedaan:</strong><br>
                    • <em>Make</em> = memaksa/menyebabkan<br>
                    • <em>Have</em> = menyuruh/meminta<br>
                    • <em>Get</em> = membujuk/meyakinkan
                </div>
            </div>

    <script>
        // Quick Quiz Functionality
        let selectedOption = null;

        document.querySelectorAll('.quiz-option').forEach(option => {
            option.addEventListener('click', function() {
                document.querySelectorAll('.quiz-option').forEach(opt => {
                    opt.classList.remove('selected');
                });
                this.classList.add('selected');
                selectedOption = this.getAttribute('data-value');
            });
        });

        function checkQuiz() {
            const resultDiv = document.getElementById('quiz-result');
            
            if (!selectedOption) {
                resultDiv.textContent = "Pilih jawaban terlebih dahulu!";
                resultDiv.className = "quiz-result incorrect";
                resultDiv.style.display = 'block';
                return;
            }

            if (selectedOption === 'A') {
                resultDiv.innerHTML = "✅ <strong>Benar!</strong><br>Present Simple untuk kebiasaan (goes) dan Present Continuous untuk kegiatan sekarang (is going)";
                resultDiv.className = "quiz-result correct";
            } else {
                resultDiv.innerHTML = "❌ <strong>Masih salah!</strong><br>Jawaban yang benar: A. goes / is going<br>Present Simple untuk kebiasaan sehari-hari, Present Continuous untuk kegiatan spesifik sekarang";
                resultDiv.className = "quiz-result incorrect";
            }
            
            resultDiv.style.display = 'block';
        }

        // Smooth scrolling for table of contents
        document.querySelectorAll('.toc-list a').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                const targetElement = document.querySelector(targetId);
                targetElement.scrollIntoView({ behavior: 'smooth' });
            });
        });
    </script>
