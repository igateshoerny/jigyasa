                <h2>⚡ Skimming & Scanning Techniques</h2>
                
                <h3>Skimming - Membaca Cepat untuk Ide Utama</h3>
                <div class="strategy-box">
                    <strong>Kapan digunakan:</strong> Untuk memahami main idea/gist<br>
                    <strong>Caranya:</strong><br>
                    • Baca judul dan subjudul<br>
                    • Baca kalimat pertama setiap paragraf<br>
                    • Baca kalimat terakhir (kesimpulan)<br>
                    • Abaikan detail dan contoh
                </div>

                <h3>Scanning - Mencari Informasi Spesifik</h3>
                <div class="strategy-box">
                    <strong>Kapan digunakan:</strong> Untuk mencari fakta/detail tertentu<br>
                    <strong>Caranya:</strong><br>
                    • Cari kata kunci (names, dates, numbers)<br>
                    • Gunakan jari atau pointer<br>
                    • Lewati informasi tidak relevan<br>
                    • Fokus pada kata yang dicari
                </div>

                <div class="tip-box">
                    <strong>💡 Tips Olimpiade:</strong><br>
                    • <strong>Skimming</strong> dulu untuk memahami struktur teks<br>
                    • Baca <strong>pertanyaan</strong> sebelum membaca teks detail<br>
                    • Gunakan <strong>scanning</strong> untuk jawaban detail questions
                </div>
            </div>

        // Function for individual answer checking
        function checkAnswer(questionId, correctAnswer) {
            const options = document.querySelectorAll(`#${questionId} .quiz-option`);
            const resultDiv = document.getElementById(`${questionId}-result`);
            
            let selectedOption = null;
            options.forEach(option => {
                if (option.classList.contains('selected')) {
                    selectedOption = option.getAttribute('data-value');
                }
            });

            if (!selectedOption) {
                resultDiv.textContent = "Pilih jawaban terlebih dahulu!";
                resultDiv.className = "quiz-result incorrect";
                resultDiv.style.display = 'block';
                return;
            }

            if (selectedOption === correctAnswer) {
                resultDiv.innerHTML = "✅ <strong>Benar!</strong>";
                resultDiv.className = "quiz-result correct";
            } else {
                resultDiv.innerHTML = `❌ <strong>Masih salah!</strong> Jawaban benar: ${correctAnswer}`;
                resultDiv.className = "quiz-result incorrect";
            }
            
            resultDiv.style.display = 'block';
        }

            let allAnswered = true;
            let score = 0;
            let feedback = "";

            for (let i = 0; i < practiceQuestions.length; i++) {
                const options = document.querySelectorAll(`#practice .quiz-options`)[i].querySelectorAll('.quiz-option');
                let selectedOption = null;

                options.forEach(option => {
                    if (option.classList.contains('selected')) {
                        selectedOption = option.getAttribute('data-value');
                    }
                });

                if (!selectedOption) {
                    allAnswered = false;
                    break;
                }

                if (selectedOption === practiceQuestions[i].correct) {
                    score++;
                    feedback += `✅ Pertanyaan ${i+1}: Benar! ${practiceQuestions[i].explanation}<br>`;
                } else {
                    feedback += `❌ Pertanyaan ${i+1}: Salah! Jawaban benar: ${practiceQuestions[i].correct}. ${practiceQuestions[i].explanation}<br>`;
                }
            }

            const resultDiv = document.getElementById('practice-result');
            
            if (!allAnswered) {
                resultDiv.textContent = "Jawab semua pertanyaan terlebih dahulu!";
                resultDiv.className = "quiz-result incorrect";
                resultDiv.style.display = 'block';
                return;
            }

            resultDiv.innerHTML = `<strong>Score: ${score}/${practiceQuestions.length}</strong><br><br>${feedback}`;
            resultDiv.className = score === practiceQuestions.length ? "quiz-result correct" : "quiz-result incorrect";
            resultDiv.style.display = 'block';
        }

        // Add click event to all quiz options
        document.querySelectorAll('.quiz-option').forEach(option => {
            option.addEventListener('click', function() {
                // Remove selection from other options in the same question
                const parentOptions = this.closest('.quiz-options');
                parentOptions.querySelectorAll('.quiz-option').forEach(opt => {
                    opt.classList.remove('selected');
                });
                
                this.classList.add('selected');
            });
        });

        // Smooth scrolling for table of contents
        document.querySelectorAll('.toc-list a').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                const targetElement = document.querySelector(targetId);
                targetElement.scrollIntoView({ behavior: 'smooth' });
            });
        });
    </script>
</body>
